*/
function recebe_tres(x, y, z){
    if (typeof x === "number" && isNaN(y) && isNaN(z)) {
        console.log("x = ", x)
    } else if (typeof x === "number" && typeof y === "number" && isNaN(z)) {
        console.log("x + y = ", x + y)
    } else if (typeof x === "number" && typeof y === "number" && typeof z === "number"){

    } else {

    } 
}
